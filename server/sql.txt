-- 4. AUTOMATIC PROFILE CREATION TRIGGER
--    Crea automáticamente un perfil en public.profiles cuando un
--    usuario se registra en auth.users (vía Email o Google).
--    Ejecutar esto en el SQL Editor de Supabase.

CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
DECLARE
  username_val TEXT;
  display_name_val TEXT;
  provider_val TEXT;
  photo_url_val TEXT;
BEGIN
  -- Obtener el nombre de usuario desde los metadatos o fallback a email
  username_val := COALESCE(
    NEW.raw_user_meta_data->>'username',
    NEW.raw_user_meta_data->>'full_name',
    NEW.raw_user_meta_data->>'name',
    split_part(NEW.email, '@', 1)
  );
  display_name_val := COALESCE(
    NEW.raw_user_meta_data->>'full_name',
    NEW.raw_user_meta_data->>'name',
    username_val
  );
  photo_url_val := COALESCE(
    NEW.raw_user_meta_data->>'avatar_url',
    NEW.raw_user_meta_data->>'picture',
    ''
  );
  -- Determinar el proveedor
  IF NEW.raw_app_meta_data->>'provider' IS NOT NULL THEN
    provider_val := NEW.raw_app_meta_data->>'provider';
  ELSE
    provider_val := 'email';
  END IF;
  INSERT INTO public.profiles (id, username, display_name, email, photo_url, provider)
  VALUES (
    NEW.id,
    username_val,
    display_name_val,
    NEW.email,
    photo_url_val,
    provider_val
  )
  ON CONFLICT (id) DO UPDATE
  SET
    username = EXCLUDED.username,
    display_name = EXCLUDED.display_name,
    email = EXCLUDED.email,
    photo_url = EXCLUDED.photo_url,
    provider = EXCLUDED.provider,
    updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- ────────────────────────────────────────────────────────────
-- 10. FUNCIÓN: Sincronizar progress_keys con user_progress
-- ────────────────────────────────────────────────────────────
-- Esta función se ejecuta cada vez que se actualiza progress_keys
-- y actualiza automáticamente user_progress con el resumen
CREATE OR REPLACE FUNCTION sync_progress_keys_to_user_progress()
RETURNS TRIGGER AS $$
DECLARE
  v_count INT;
  v_total INT;
  v_percentage FLOAT;
BEGIN
  -- Contar cuántos pkey=true hay
  SELECT COUNT(*) INTO v_count
  FROM progress_keys
  WHERE user_id = NEW.user_id 
    AND category = NEW.category 
    AND item_id = NEW.item_id 
    AND value = true;

  -- Determinar el total basado en la categoría
  -- (Esto es aproximado, idealmente obtendrías esto de la API)
  v_total := v_count;
  
  -- Si no hay registros completados, asignar 0%
  IF v_count = 0 THEN
    v_percentage := 0;
  ELSE
    -- Calcular porcentaje (asumiendo que contamos los completados)
    v_percentage := (v_count::FLOAT / NULLIF(v_total, 0)) * 100;
  END IF;

  -- Upsert en user_progress
  INSERT INTO user_progress (
    user_id, category, item_id, 
    episodes_viewed, progress_percentage, updated_at
  )
  VALUES (
    NEW.user_id, NEW.category, NEW.item_id,
    v_count, v_percentage, NOW()
  )
  ON CONFLICT (user_id, category, item_id) 
  DO UPDATE SET 
    episodes_viewed = v_count,
    progress_percentage = v_percentage,
    updated_at = NOW();

  -- Registrar en activity log
  INSERT INTO user_activity_log (user_id, action, category, item_id, value, created_at)
  VALUES (NEW.user_id, 'progress_updated', NEW.category, NEW.item_id, v_count, NOW());

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS sync_progress_keys_trigger ON progress_keys;
CREATE TRIGGER sync_progress_keys_trigger
AFTER INSERT OR UPDATE ON progress_keys
FOR EACH ROW
EXECUTE FUNCTION sync_progress_keys_to_user_progress();

CREATE OR REPLACE FUNCTION public.rebuild_user_progress_row(
  p_user_id UUID,
  p_category TEXT,
  p_item_id TEXT
)
RETURNS VOID AS $$
DECLARE
  v_episodes INT := 0;
  v_chapters INT := 0;
  v_volumes INT := 0;
  v_total_episodes INT;
  v_total_chapters INT;
  v_total_volumes INT;
  v_total INT := 0;
  v_done INT := 0;
  v_percentage FLOAT := 0;
BEGIN
  SELECT COUNT(*) INTO v_episodes
  FROM public.progress_keys
  WHERE user_id = p_user_id
    AND category = p_category
    AND item_id = p_item_id
    AND value = true
    AND pkey ~ '^s:[0-9]+\|ep:[0-9]+$';

  SELECT COUNT(*) INTO v_chapters
  FROM public.progress_keys
  WHERE user_id = p_user_id
    AND category = p_category
    AND item_id = p_item_id
    AND value = true
    AND pkey ~ '^ch:[0-9]+$';

  SELECT COUNT(*) INTO v_volumes
  FROM public.progress_keys
  WHERE user_id = p_user_id
    AND category = p_category
    AND item_id = p_item_id
    AND value = true
    AND pkey ~ '^vol:[0-9]+$';

  SELECT total_episodes, total_chapters, total_volumes
  INTO v_total_episodes, v_total_chapters, v_total_volumes
  FROM public.user_progress
  WHERE user_id = p_user_id
    AND category = p_category
    AND item_id = p_item_id;

  IF p_category = 'anime' THEN
    v_done := v_episodes;
    v_total := COALESCE(v_total_episodes, 0);
  ELSIF p_category IN ('manga', 'novelas') THEN
    IF v_volumes > 0 THEN
      v_done := v_volumes;
      v_total := COALESCE(v_total_volumes, 0);
    ELSE
      v_done := v_chapters;
      v_total := COALESCE(v_total_chapters, 0);
    END IF;
  ELSE
    v_done := GREATEST(v_episodes, v_chapters, v_volumes);
    v_total := 0;
  END IF;

  IF v_total > 0 THEN
    v_percentage := LEAST(100, ROUND(((v_done::NUMERIC / v_total::NUMERIC) * 100), 2)::FLOAT);
  ELSE
    v_percentage := 0;
  END IF;

  IF v_episodes = 0 AND v_chapters = 0 AND v_volumes = 0 THEN
    DELETE FROM public.user_progress
    WHERE user_id = p_user_id
      AND category = p_category
      AND item_id = p_item_id;
    RETURN;
  END IF;

  INSERT INTO public.user_progress (
    user_id,
    category,
    item_id,
    episodes_viewed,
    chapters_viewed,
    volumes_viewed,
    total_episodes,
    total_chapters,
    total_volumes,
    progress_percentage,
    updated_at
  )
  VALUES (
    p_user_id,
    p_category,
    p_item_id,
    v_episodes,
    v_chapters,
    v_volumes,
    v_total_episodes,
    v_total_chapters,
    v_total_volumes,
    v_percentage,
    NOW()
  )
  ON CONFLICT (user_id, category, item_id)
  DO UPDATE SET
    episodes_viewed = EXCLUDED.episodes_viewed,
    chapters_viewed = EXCLUDED.chapters_viewed,
    volumes_viewed = EXCLUDED.volumes_viewed,
    total_episodes = COALESCE(public.user_progress.total_episodes, EXCLUDED.total_episodes),
    total_chapters = COALESCE(public.user_progress.total_chapters, EXCLUDED.total_chapters),
    total_volumes = COALESCE(public.user_progress.total_volumes, EXCLUDED.total_volumes),
    progress_percentage = EXCLUDED.progress_percentage,
    updated_at = NOW();
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE OR REPLACE FUNCTION public.sync_progress_keys_to_user_progress()
RETURNS TRIGGER AS $$
DECLARE
  v_user_id UUID;
  v_category TEXT;
  v_item_id TEXT;
  v_value INT;
BEGIN
  v_user_id := COALESCE(NEW.user_id, OLD.user_id);
  v_category := COALESCE(NEW.category, OLD.category);
  v_item_id := COALESCE(NEW.item_id, OLD.item_id);

  PERFORM public.rebuild_user_progress_row(v_user_id, v_category, v_item_id);

  SELECT COALESCE(
    CASE
      WHEN v_category = 'anime' THEN episodes_viewed
      WHEN v_category IN ('manga', 'novelas') THEN GREATEST(chapters_viewed, volumes_viewed)
      ELSE GREATEST(episodes_viewed, chapters_viewed, volumes_viewed)
    END,
    0
  )
  INTO v_value
  FROM public.user_progress
  WHERE user_id = v_user_id
    AND category = v_category
    AND item_id = v_item_id;

  INSERT INTO public.user_activity_log (user_id, action, category, item_id, value, created_at)
  VALUES (v_user_id, 'progress_updated', v_category, v_item_id, COALESCE(v_value, 0), NOW());

  IF TG_OP = 'DELETE' THEN
    RETURN OLD;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER sync_progress_keys_trigger

EXECUTE FUNCTION public.sync_progress_keys_to_user_progress();

-- 2. Arreglamos el Trigger de estadísticas para que reste al hacer DELETE
CREATE OR REPLACE FUNCTION update_user_stats_on_item_change()
RETURNS TRIGGER AS $$
DECLARE
  v_total_likes INT;
  v_total_viewed INT;
  v_user_id UUID;
BEGIN
  -- Usamos COALESCE para soportar tanto INSERT como DELETE
  v_user_id := COALESCE(NEW.user_id, OLD.user_id);

  SELECT COUNT(*) INTO v_total_likes
  FROM item_states
  WHERE user_id = v_user_id AND fav = true;

  SELECT COUNT(*) INTO v_total_viewed
  FROM item_states
  WHERE user_id = v_user_id AND viewed = true;

  UPDATE profiles
  SET total_likes = v_total_likes, total_viewed = v_total_viewed, updated_stats_at = NOW()
  WHERE id = v_user_id;

  IF TG_OP = 'DELETE' THEN
    RETURN OLD;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS update_user_stats_trigger ON item_states;
CREATE TRIGGER update_user_stats_trigger
AFTER INSERT OR UPDATE OR DELETE ON item_states
FOR EACH ROW
EXECUTE FUNCTION update_user_stats_on_item_change();

-- 3. Arreglamos el Trigger de actividad para registrar cuando se quita un item
CREATE OR REPLACE FUNCTION log_item_state_activity()
RETURNS TRIGGER AS $$
DECLARE
  v_user_id UUID;
  v_category TEXT;
  v_item_id TEXT;
  v_action TEXT;
BEGIN
  IF TG_OP = 'DELETE' THEN
    v_user_id := OLD.user_id;
    v_category := OLD.category;
    v_item_id := OLD.item_id;
    v_action := 'removed';
  ELSE
    v_user_id := NEW.user_id;
    v_category := NEW.category;
    v_item_id := NEW.item_id;
    v_action := CASE 
      WHEN NEW.fav = true THEN 'liked'
      WHEN NEW.viewed = true THEN 'viewed'
      ELSE 'item_state_changed'
    END;
  END IF;

  INSERT INTO user_activity_log (user_id, action, category, item_id, created_at)
  VALUES (v_user_id, v_action, v_category, v_item_id, NOW());

  IF TG_OP = 'DELETE' THEN
    RETURN OLD;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 2. Función para sumar XP (la llama el JS)
CREATE OR REPLACE FUNCTION add_user_exp(p_user_id UUID, p_delta INT)
RETURNS void AS $$
DECLARE
  v_new_exp INT;
  v_new_level INT;
BEGIN
  UPDATE profiles
  SET exp = GREATEST(0, exp + p_delta)
  WHERE id = p_user_id
  RETURNING exp INTO v_new_exp;

  -- Calcular nivel (misma lógica que el JS)
  v_new_level := 1;
  DECLARE need INT := 100; remaining INT := v_new_exp;
  WHILE remaining >= need AND v_new_level < 50 LOOP
    remaining := remaining - need;
    v_new_level := v_new_level + 1;
    need := FLOOR(need * 1.2);
  END LOOP;

  UPDATE profiles SET level = v_new_level WHERE id = p_user_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

    VALUES (NEW.user_id, 'viewed_added', NEW.category, NEW.item_id);
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
